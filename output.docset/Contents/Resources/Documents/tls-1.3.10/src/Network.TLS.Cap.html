module Network.TLS.Cap
    ( hasHelloExtensions
    , hasExplicitBlockIV
    ) where

import Network.TLS.Struct

hasHelloExtensions, hasExplicitBlockIV :: Version -> Bool

hasHelloExtensions ver = ver >= SSL3
hasExplicitBlockIV ver = ver >= TLS11
